# 2. AGORA, ATUALIZE A CONFIGURA√á√ÉO DO NGINX (O "PORTEIRO"):
# Copie TODO o bloco abaixo de uma vez s√≥ e cole no terminal, depois aperte ENTER.
# Isso vai reescrever o arquivo de configura√ß√£o com o dom√≠nio novo e as regras certas.

cat <<EOF | sudo tee /etc/nginx/sites-available/audioloop
server {
    listen 80;
    server_name api.audioloop.com.br;
    return 301 https://\$host\$request_uri;
}

server {
    listen 443 ssl;
    server_name api.audioloop.com.br;

    ssl_certificate /etc/letsencrypt/live/api.audioloop.com.br/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api.audioloop.com.br/privkey.pem;

    client_max_body_size 500M;

    location / {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade \$http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host \$host;
        proxy_cache_bypass \$http_upgrade;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;

        # CORS (Permite que seu site acesse a API)
        add_header 'Access-Control-Allow-Origin' '*' always;
        add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS, PUT, DELETE' always;
        add_header 'Access-Control-Allow-Headers' 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization' always;

        if (\$request_method = 'OPTIONS') {
            add_header 'Access-Control-Allow-Origin' '*';
            add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS, PUT, DELETE';
            add_header 'Access-Control-Allow-Headers' 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization';
            add_header 'Access-Control-Max-Age' 1728000;
            add_header 'Content-Type' 'text/plain; charset=utf-8';
            add_header 'Content-Length' 0;
            return 204;
        }
    }
}
EOF
